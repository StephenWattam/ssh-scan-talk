\section{Usage}
\frame{\frametitle{Usage}
Use the style like any other beamer theme.  Simply place it in a TeX path and use:

    \vspace{10pt}
\texttt{usetheme[option,option2=value]{SmartSerif}}
    \vspace{10pt}

Options are listed in subsequent pages of this manual.
}

\subsection{Options}
\frame{\frametitle{Colour Options}
\begin{itemize}
    \item \textbf{\texttt{bannerbg=colour}} --- Set the banner background 
    \item \textbf{\texttt{bannerfg=colour}} --- Set the banner foreground
    \item \textbf{\texttt{bannerline=colour}} --- Set the colour of the line at the bottom of the banner
    \item \textbf{\texttt{curvecolour=colour}} --- Set the colour of the Koch curve in the background
    \item \textbf{\texttt{infolabcolours}} --- Uses infolab colour schemes, and turns off the curve (use \texttt{forcecurve} to turn it back on again)
    % \item \textbf{\texttt{}} --- STUFF
\end{itemize}
}


\frame{\frametitle{Display Options}
\begin{itemize}
    \item \textbf{\texttt{nocurve}} --- Disable the background curve
    \item \textbf{\texttt{forcecurve}} --- Turn the curve on when disabled by the \texttt{infolabcolours} option
    \item \textbf{\texttt{num}} --- Show slide numbers in the footer
\end{itemize}
}
